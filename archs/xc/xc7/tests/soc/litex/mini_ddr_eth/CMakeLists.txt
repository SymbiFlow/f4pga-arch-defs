  top.xdc

  mem_1.init
  mem_2.init
)

add_litex_test(
  NAME minilitex_ddr_eth_arty
  LITEX_DIR arty_soc
  LITEX_BOARD a7-35
  LITEX_SOURCES ${SOURCES}
  EXTERNAL_SOURCES ${VEXRISCV}
  BOARD arty-full
  GENERATE_SCRIPT ${GENERATE_LITEX}

)
